# netology
Решение домашки

В директории с наименованием даты будут находиться все скрипты по заданию.
Для проверки необходимо выполнить git clone в рабочую директорию.

# Для некоторых скриптов идёт поясления
# Так же каждый скрипт, который просит пользовательского ввода оборудован проверкой и выводом подсказки 
cd 2025.12.08

# countdown
./countdown.sh - необходимо передать число на вход
./countdown.sh 10

# read_file 
./read_file.sh - необходимо передать 1 аргумент (<путь_к_файлу>)
./read_file.sh dummy.txt

# sum
./sum.sh - необходимо передать два числа
./sum.sh 100 + 1

# replace_text
./replace_text.sh - необходимо передать 3 аргумента (<путь/имя_к_файлу> <искомое_слово> <заменяемое_слово>)
./replace_text.sh dummy.txt fix price
# Если работаете на МакОС, то внутри файла внёс описание

# backup
./backup.sh - необходимо передать 2 аргумента (<исходная_директория> <целевая_директория>)
./backup.sh backupMe dest

# math_operations
./math_operations.sh - подобие калькулятора, всё интерактивно
